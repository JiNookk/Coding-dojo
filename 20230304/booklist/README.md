### 1. 이해
- 조인을 이용해서 테이블을 묶기
- 조인 사용방법(INNER JOIN)
SELECT <열 목록>
FROM <기준 테이블>
    INNER JOIN<참조할 테이블>
    ON <조인 조건>
[WHERE 검색조건]

- 날짜 포맷 지정
date_format(datetime, '%Y-%m-%d')

- 정렬 방법
SELECT * FROM 테이블 WHERE 조건식 ORDER BY 컬럼1 [, 컬럼2, 컬럼3 ...];
